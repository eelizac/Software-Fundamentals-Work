78@defines_strategy(force_reusable_values=True) 

79def ip_addresses( 

80 *, 

81 v: Optional[int] = None, 

82 network: Optional[Union[str, IPv4Network, IPv6Network]] = None, 

83) -> SearchStrategy[Union[IPv4Address, IPv6Address]]: 

84 r"""Generate IP addresses - ``v=4`` for :class:`~python:ipaddress.IPv4Address`\ es, 

85 ``v=6`` for :class:`~python:ipaddress.IPv6Address`\ es, or leave unspecified 

86 to allow both versions. 

87 

88 ``network`` may be an :class:`~python:ipaddress.IPv4Network` or 

89 :class:`~python:ipaddress.IPv6Network`, or a string representing a network such as 

90 ``"127.0.0.0/24"`` or ``"2001:db8::/32"``. As well as generating addresses within 

91 a particular routable network, this can be used to generate addresses from a 

92 reserved range listed in the 

93 `IANA <https://www.iana.org/assignments/iana-ipv4-special-registry/>`__ 

94 `registries <https://www.iana.org/assignments/iana-ipv6-special-registry/>`__. 

95 

96 If you pass both ``v`` and ``network``, they must be for the same version. 

97 """ 

98 if v is not None: 

99 check_type(int, v, "v") 

100 if v not in (4, 6): 

101 raise InvalidArgument(f"v={v!r}, but only v=4 or v=6 are valid") 

102 if network is None: 

103 # We use the reserved-address registries to boost the chance 

104 # of generating one of the various special types of address. 

105 four = binary(min_size=4, max_size=4).map(IPv4Address) | sampled_from( 

106 SPECIAL_IPv4_RANGES 

107 ).flatmap(lambda network: ip_addresses(network=network)) 

108 six = binary(min_size=16, max_size=16).map(IPv6Address) | sampled_from( 

109 SPECIAL_IPv6_RANGES 

110 ).flatmap(lambda network: ip_addresses(network=network)) 

111 if v == 4: 

112 return four 

113 if v == 6: 

114 return six 

115 return four | six 

116 if isinstance(network, str): 

117 network = ip_network(network) 

118 check_type((IPv4Network, IPv6Network), network, "network") 

119 assert isinstance(network, (IPv4Network, IPv6Network)) # for Mypy 

120 if v not in (None, network.version): 

121 raise InvalidArgument(f"v={v!r} is incompatible with network={network!r}") 

122 addr_type = IPv4Address if network.version == 4 else IPv6Address 

123 return integers(int(network[0]), int(network[-1])).map(addr_type) # type: ignore
